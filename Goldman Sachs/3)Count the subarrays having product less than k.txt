//User function Template for Java


class Solution {
    
    public int countSubArrayProductLessThanK(long nums[], long n, long k)
    {
        int l=0;
        int res=0;
        long pro=1;
        for(int r=0;r<nums.length;r++)
        {
            pro=pro*nums[r];
            while(pro>=k && r>=l )
            {
                pro=pro/nums[l];
                l++;
            }
             res=res+r-l+1;
        }
        return res;
    }
}
